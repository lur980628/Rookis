def print_books_over_20000(df):
    df_filtered = df[df['discount'].astype(int) >= 20000]
    df_sorted = df_filtered[['title', 'author', 'discount', 'publisher', 'pubdate']] \
        .sort_values(by='discount', ascending=False).reset_index(drop=True)
    print(df_sorted)
